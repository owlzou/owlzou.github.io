function readFiles(fpath, fname) {
  let fn = path.resolve(fpath, fname);
  if (fs.statSync(fn).isFile() && path.extname(fn) == ".md") {
    let data = fs.readFileSync(fn, "utf8").toString().split("---");
    let frontMatter = data[1];
    let yaml = YAML.parse(frontMatter);
    Object.keys(yaml).forEach((e) => {
      if (original.has(e)) {
      } else if (taxonomies.has(e)) {
        yaml.taxonomies || (yaml.taxonomies = {});
        yaml.taxonomies[e] = yaml[e];
        delete yaml[e];
      } else {
        yaml.extra || (yaml.extra = {});
        yaml.extra[e] = yaml[e];
        delete yaml[e];
      }
    });
    let toml = TOML.stringify(yaml);
    toml = toml
      .replace(/(date|updated) = "(\S*) +(\S*)"/g, "$1 = $2T$3Z")
      .replace(/(date|updated) = "(.+)"/g, "$1 = $2");
    data[1] = `+++\n${toml}+++`;
    file = data.join("").replace("<!--more-->","<!-- more -->");
    fs.writeFileSync(path.resolve(output_path, fname), file);
  }
}
